=Usage=
Used to specify what keepers and adventurers appear in games and also what immigrants appear in games.

=Example from player_creatures.txt=
 creatureId = { KEEPER_MAGE KEEPER_MAGE_F }

=Example from immigration.txt=
 ids = { IMP }
